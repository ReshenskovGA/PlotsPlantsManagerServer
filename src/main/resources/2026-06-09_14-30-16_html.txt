<script th:inline="javascript">
    function formatDateLocal(timestamp) {
        var d = new Date(timestamp);
        var year = d.getFullYear();
        var month = ('0' + (d.getMonth() + 1)).slice(-2);
        var day = ('0' + d.getDate()).slice(-2);
        return year + '-' + month + '-' + day;
    }

    document.addEventListener('DOMContentLoaded', function() {
        var calendarEl = document.getElementById('calendar');
        var tasks = /*[[${tasks}]]*/ [];
        var toggleUrlBase = /*[[@{/web/tasks/toggle}]]*/ '/web/tasks/toggle';
        var dayUrlBase = /*[[@{/web/tasks/day}]]*/ '/web/tasks/day';

        // Получаем CSRF-токен из meta-тега
        var csrfToken = document.querySelector('meta[name="_csrf"]').getAttribute('content');

        var today = new Date();
        today.setHours(0, 0, 0, 0);
        var tomorrow = new Date(today);
        tomorrow.setDate(tomorrow.getDate() + 1);

        var events = tasks.map(function(task) {
            var taskDate = new Date(task.date);
            taskDate.setHours(0, 0, 0, 0);

            var bgColor, borderColor, textColor;
            if (task.isCompleted) {
                bgColor = '#4caf50';
                borderColor = '#388e3c';
                textColor = '#fff';
            } else if (taskDate.getTime() < today.getTime()) {
                bgColor = '#ef5350';
                borderColor = '#c62828';
                textColor = '#fff';
            } else if (taskDate.getTime() < tomorrow.getTime()) {
                bgColor = '#ffee58';
                borderColor = '#f9a825';
                textColor = '#333';
            } else {
                bgColor = '#42a5f5';
                borderColor = '#1565c0';
                textColor = '#fff';
            }

            return {
                title: task.title + (task.isCompleted ? ' (Вып.)' : ''),
                start: formatDateLocal(task.date),
                backgroundColor: bgColor,
                borderColor: borderColor,
                textColor: textColor,
                extendedProps: {
                    id: task.id,
                    date: task.date,
                    description: task.description,
                    isCompleted: task.isCompleted
                }
            };
        });

        var calendar = new FullCalendar.Calendar(calendarEl, {
            initialView: 'dayGridMonth',
            locale: 'ru',
            headerToolbar: {
                left: 'prev,next today',
                center: 'title',
                right: 'dayGridMonth,timeGridWeek,listMonth'
            },
            events: events,
            eventClick: function(info) {
                info.jsEvent.preventDefault();
                var actionText = info.event.extendedProps.isCompleted ? 'невыполненную' : 'выполненную';
                if (confirm('Отметить задачу "' + info.event.title.replace(' (Вып.)', '') + '" как ' + actionText + '?')) {
                    var form = document.createElement('form');
                    form.method = 'POST';
                    form.action = toggleUrlBase;

                    var idInput = document.createElement('input');
                    idInput.type = 'hidden';
                    idInput.name = 'id';
                    idInput.value = info.event.extendedProps.id;
                    form.appendChild(idInput);

                    var dateInput = document.createElement('input');
                    dateInput.type = 'hidden';
                    dateInput.name = 'date';
                    dateInput.value = info.event.extendedProps.date;
                    form.appendChild(dateInput);

                    // === ИСПРАВЛЕНИЕ: добавляем CSRF-токен ===
                    var csrfInput = document.createElement('input');
                    csrfInput.type = 'hidden';
                    csrfInput.name = '_csrf';
                    csrfInput.value = csrfToken;
                    form.appendChild(csrfInput);

                    document.body.appendChild(form);
                    form.submit();
                }
            },
            dateClick: function(info) {
                window.location.href = dayUrlBase + '?date=' + info.dateStr;
            }
        });
        calendar.render();
    });
</script>

<script th:inline="javascript">
    document.addEventListener('DOMContentLoaded', function() {
        const enableRecurrence = document.getElementById('enableRecurrence');
        const recurrenceSettings = document.getElementById('recurrenceSettings');
        const recurrenceType = document.getElementById('recurrenceType');
        const weeklyDays = document.getElementById('weeklyDays');
        const endType = document.getElementById('endType');
        const endDateContainer = document.getElementById('endDateContainer');
        const endCountContainer = document.getElementById('endCountContainer');
        const recurrenceRuleJsonInput = document.getElementById('recurrenceRuleJsonInput');

        // Если есть существующее правило, распарсим его и заполним форму
        const existingRuleJson = recurrenceRuleJsonInput.value;
        if (existingRuleJson && existingRuleJson.trim() !== '') {
            try {
                const rule = JSON.parse(existingRuleJson);
                if (rule.type && rule.type !== 'NONE') {
                    enableRecurrence.checked = true;
                    recurrenceSettings.style.display = 'block';
                    recurrenceType.value = rule.type;
                    if (rule.interval) document.getElementById('recurrenceInterval').value = rule.interval;

                    if (rule.type === 'WEEKLY' && rule.weekDays) {
                        weeklyDays.style.display = 'block';
                        rule.weekDays.forEach(day => {
                            const cb = document.querySelector(`.week-day[value="${day}"]`);
                            if (cb) cb.checked = true;
                        });
                    }

                    if (rule.endType) {
                        endType.value = rule.endType;
                        if (rule.endType === 'DATE' && rule.endDate) {
                            const d = new Date(rule.endDate);
                            document.getElementById('endDate').value = d.toISOString().split('T')[0];
                        } else if (rule.endType === 'COUNT' && rule.endCount) {
                            document.getElementById('endCount').value = rule.endCount;
                        }
                    }
                }
            } catch (e) {
                console.error("Ошибка парсинга правила повторения", e);
            }
        }

        function updateVisibility() {
            recurrenceSettings.style.display = enableRecurrence.checked ? 'block' : 'none';
            weeklyDays.style.display = recurrenceType.value === 'WEEKLY' ? 'block' : 'none';
            endDateContainer.style.display = endType.value === 'DATE' ? 'block' : 'none';
            endCountContainer.style.display = endType.value === 'COUNT' ? 'block' : 'none';
            generateRuleJson();
        }

        function generateRuleJson() {
            if (!enableRecurrence.checked) {
                recurrenceRuleJsonInput.value = '';
                return;
            }

            const rule = {
                type: recurrenceType.value,
                interval: parseInt(document.getElementById('recurrenceInterval').value) || 1,
                endType: endType.value
            };

            if (rule.type === 'WEEKLY') {
                const days = [];
                document.querySelectorAll('.week-day:checked').forEach(cb => days.push(parseInt(cb.value)));
                if (days.length > 0) rule.weekDays = days;
            }

            if (rule.endType === 'DATE') {
                const dateVal = document.getElementById('endDate').value;
                if (dateVal) {
                    rule.endDate = new Date(dateVal).getTime();
                }
            } else if (rule.endType === 'COUNT') {
                rule.endCount = parseInt(document.getElementById('endCount').value) || 1;
            }

            recurrenceRuleJsonInput.value = JSON.stringify(rule);
        }

        enableRecurrence.addEventListener('change', updateVisibility);
        recurrenceType.addEventListener('change', updateVisibility);
        endType.addEventListener('change', updateVisibility);
        document.querySelectorAll('.week-day').forEach(cb => cb.addEventListener('change', generateRuleJson));
        document.getElementById('recurrenceInterval').addEventListener('input', generateRuleJson);
        document.getElementById('endDate').addEventListener('change', generateRuleJson);
        document.getElementById('endCount').addEventListener('input', generateRuleJson);

        // Инициализация при загрузке
        updateVisibility();

        // Перед отправкой формы убедимся, что JSON актуален
        document.querySelector('form').addEventListener('submit', function(e) {
            generateRuleJson();
        });
    });
</script>
